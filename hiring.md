<script src="https://cdn.jsdelivr.net/npm/marked/marked.min.js"></script>
<script>
  const form = document.getElementById('feedbackForm');
  const resumeEl = document.getElementById('resume');
  const previewEl = document.getElementById('preview');
  const messageDiv = document.getElementById('formMessage');
  const submitBtn = document.getElementById('submitBtn');

  // Live markdown preview
  resumeEl.addEventListener('input', function() {
    const markdown = this.value;
    previewEl.innerHTML = marked.parse(markdown);
  });

  // Initialize preview with placeholder
  previewEl.innerHTML = marked.parse(resumeEl.placeholder);

  form.addEventListener('submit', function(e) {
    e.preventDefault();

    const environment   = document.getElementById('environment').value.trim();
    const support       = document.getElementById('support').value.trim();
    const tools         = document.getElementById('tools').value.trim();
    const growth        = document.getElementById('growth').value.trim();
    const anythingElse  = document.getElementById('anything-else').value.trim();
    const resume        = resumeEl.value.trim();

    if (!resume) {
      alert('Please provide your resume in Markdown.');
      return;
    }

    submitBtn.disabled = true;
    submitBtn.textContent = 'Preparing your application…';

    let bodyLines = [];

    bodyLines.push('## Work Environment');
    bodyLines.push(environment || '_Not provided_');
    bodyLines.push('');

    bodyLines.push('## Support from Leadership');
    bodyLines.push(support || '_Not provided_');
    bodyLines.push('');

    bodyLines.push('## First 30 Days – Tools & Resources');
    bodyLines.push(tools || '_Not provided_');
    bodyLines.push('');

    if (growth) {
      bodyLines.push('## Professional Growth');
      bodyLines.push(growth);
      bodyLines.push('');
    }

    if (anythingElse) {
      bodyLines.push('## Anything Else');
      bodyLines.push(anythingElse);
      bodyLines.push('');
    }

    bodyLines.push('---');
    bodyLines.push('## Resume (Markdown)');
    bodyLines.push('');
    bodyLines.push(resume || '_No resume provided_');

    const issueBody = bodyLines.join('\n');

    const timestamp = new Date().toISOString();
    const title = `Hiring Application from Website – ${timestamp}`;

    const params = new URLSearchParams({
      title: title,
      body: issueBody,
      labels: 'hiring,candidate'
    });

    const githubUrl = `https://github.com/platformbuilds/platformbuilds.github.io/issues/new?${params.toString()}`;

    messageDiv.style.display = 'block';
    messageDiv.className = 'form-message info';
    messageDiv.textContent = 'You will be redirected to GitHub in a new tab. Please review your application there and click "Submit new issue" to complete it.';

    // Open GitHub issue creation page in a new tab
    window.open(githubUrl, '_blank', 'noopener,noreferrer');

    // Re-enable button for potential re-use
    submitBtn.disabled = false;
    submitBtn.textContent = 'Share Your Insights';
  });
</script>
